# --- Fixed Income ML & Strategy Mini-Pipeline (all-in-one) ---
import os, warnings
import numpy as np, pandas as pd, matplotlib.pyplot as plt
from sklearn.decomposition import PCA
from sklearn.linear_model import RidgeCV
from sklearn.model_selection import TimeSeriesSplit

warnings.filterwarnings("ignore")
plt.rcParams["figure.figsize"] = (9, 4)
DATA_DIR = "data"  # optional folder for CSVs

# ---------- helpers ----------
def month_end_index(start="2005-01-31", end="2024-12-31"):
    return pd.date_range(start, end, freq="M")

def drawdown(series: pd.Series):
    cum = series.cumsum()
    peak = cum.cummax()
    dd = cum - peak
    return dd, dd.min()

def load_panel_from_csv_folder(folder):
    frames = []
    if not os.path.isdir(folder): 
        return None
    for fn in os.listdir(folder):
        if not fn.lower().endswith(".csv"): 
            continue
        name = os.path.splitext(fn)[0]
        df = pd.read_csv(os.path.join(folder, fn))
        if "date" not in df or "value" not in df:
            continue
        df["date"] = pd.to_datetime(df["date"])
        df = df.sort_values("date").set_index("date")[["value"]].rename(columns={"value": name})
        frames.append(df)
    if not frames: 
        return None
    return pd.concat(frames, axis=1).sort_index().resample("M").last()

def synthetic_panel(idx):
    rng = np.random.default_rng(7)
    base10 = 0.025 + np.cumsum(rng.normal(0, 0.0007, len(idx)))
    base2  = base10 + rng.normal(0.0, 0.002, len(idx)) + 0.002
    base1  = base2  + rng.normal(0.0, 0.0015, len(idx)) + 0.001
    base5  = 0.5*(base2+base10) + rng.normal(0.0,0.0012,len(idx))
    base30 = base10 + rng.normal(0.0, 0.001, len(idx)) + 0.005
    fedfunds = (base1 - 0.003).clip(0, None)
    cpi_index = 260 + np.cumsum(rng.normal(0.2, 0.1, len(idx)))
    unrate = 4.5 + np.cumsum(rng.normal(0, 0.03, len(idx)))
    unrate = np.clip(unrate, 3.0, 12.0)
    return pd.DataFrame({
        "GS1": 100*base1, "GS2": 100*base2, "GS5": 100*base5, "GS10": 100*base10, "GS30": 100*base30,
        "FEDFUNDS": 100*fedfunds, "CPIAUCSL": cpi_index, "UNRATE": unrate
    }, index=idx)

def duration_neutral_weights(dur2: float, dur10: float):
    w10 = 1.0
    w2 = -(dur10/dur2)*w10
    return w2, w10

def hist_var(series: pd.Series, alpha=0.95):
    return -np.quantile(series.dropna(), 1-alpha)

def expected_shortfall(series: pd.Series, alpha=0.95):
    q = np.quantile(series.dropna(), 1-alpha)
    tail = series[series <= q]
    return -tail.mean()

# ---------- data ----------
panel = load_panel_from_csv_folder(DATA_DIR)
if panel is None:
    idx = month_end_index()
    panel = synthetic_panel(idx)

# ---------- features ----------
yields_cols = [c for c in ["GS1","GS2","GS5","GS10","GS30"] if c in panel.columns]
yields = panel[yields_cols].dropna()
pca = PCA(n_components=min(3, len(yields_cols)))
scores = pca.fit_transform(yields.values)
factors = pd.DataFrame(scores, index=yields.index, columns=[f"PC{i+1}" for i in range(pca.n_components_)])
explained = pca.explained_variance_ratio_

fe = pd.DataFrame(index=panel.index)
if "CPIAUCSL" in panel: fe["CPI_YoY"] = panel["CPIAUCSL"].pct_change(12)*100
if "UNRATE"   in panel: fe["UNRATE"]  = panel["UNRATE"]
if "FEDFUNDS" in panel:
    fe["FEDFUNDS"]   = panel["FEDFUNDS"]
    fe["dFEDFUNDS_3M"] = panel["FEDFUNDS"].diff(3)

features = pd.concat([yields, factors, fe], axis=1).dropna()

# ---------- target & model ----------
panel["slope_2s10s"] = panel["GS10"] - panel["GS2"]
target = panel["slope_2s10s"].diff(1).shift(-1)  # next-month slope
df = pd.concat([features, target.rename("d_slope_1M")], axis=1).dropna()
X, y = df.drop(columns=["d_slope_1M"]), df["d_slope_1M"]

tscv = TimeSeriesSplit(n_splits=5)
model = RidgeCV(alphas=(0.1, 1.0, 10.0), cv=tscv)
model.fit(X, y)
pred_in = pd.Series(model.predict(X), index=X.index, name="pred")

ax = y.plot(label="actual", alpha=0.7)
pred_in.plot(ax=ax, label="pred")
plt.title("Next-Month 2s10s Slope Change (in-sample)")
plt.legend(); plt.show()

print("Explained variance by PCA:", np.round(explained, 3))
print("Ridge alpha:", model.alpha_)

# ---------- DV01-neutral trade ----------
dur2, dur10 = 1.9, 8.5
w2, w10 = duration_neutral_weights(dur2, dur10)
print("Relative weights (2Y, 10Y):", w2, w10)

delta_y2  = panel["GS2"].diff(1).loc[X.index] / 100.0
delta_y10 = panel["GS10"].diff(1).loc[X.index] / 100.0
pnl = (-dur2 * w2 * delta_y2) + (-dur10 * w10 * delta_y10)
pnl = pnl.dropna()

pnl.cumsum().plot(); plt.title("DV01-Neutral 2s10s Strategy  Cumulative PnL"); plt.show()
dd, min_dd = drawdown(pnl)
dd.plot(); plt.title(f"Drawdown (min={min_dd:.2f})"); plt.show()

print("VaR(95%):", hist_var(pnl))
print("ES(95%):", expected_shortfall(pnl))
print("Mean pnl:", pnl.mean(), "Std:", pnl.std(), "Sharpe (naive):", pnl.mean()/pnl.std())

# ---------- Summary ----------
latest_date = y.index[-1].strftime("%Y-%m")
latest_pred = pred_in.iloc[-1]
tilt = "Steepen" if latest_pred>0 else "Flatten"

""")
